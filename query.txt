-- Create Table System
CREATE TABLE SYSTEM(
	NAME varchar(50) NOT NULL,
	DESCRIPTION varchar(200) NULL,
  PRIMARY KEY (NAME ))


--Create Table USER_Account
CREATE TABLE USER_ACCOUNT(ACCOUNTID varchar(3) NOT NULL,SYSTEMNAME varchar(50) NOT NULL,ENABLED Number(1) NOT NULL,
PRIMARY KEY (ACCOUNTID,SYSTEMNAME) , FOREIGN KEY (SYSTEMNAME) REFERENCES SYSTEM(NAME)) ;
